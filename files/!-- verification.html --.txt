<body>
    <h2>Qualification Verification</h2>
    <p id="verificationStatus" class="status"></p>
    
    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const qualified = localStorage.getItem('qualified');
            const statusElement = document.getElementById('verificationStatus');
            
            if (qualified === 'yes') {
                statusElement.textContent = 'Your qualification status is verified as real.';
                statusElement.style.color = 'green';
            } else {
                statusElement.textContent = 'Your qualification status could not be verified.';
                statusElement.style.color = 'red';
            }
        });
    </script>
